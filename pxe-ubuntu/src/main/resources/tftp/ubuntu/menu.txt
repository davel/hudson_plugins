menu hshift 13
menu width 49
menu margin 8

menu title Installer boot menu (${RELEASE})

include ubuntu/stdmenu.cfg

label install
	menu label Install (i^386)
	kernel ubuntu-${RELEASE}/i386/linux
	append vga=normal initrd=ubuntu-${RELEASE}/i386/initrd.gz -- quiet

label install
	menu label Install (amd^64)
	kernel ubuntu-${RELEASE}/amd64/linux
	append vga=normal initrd=ubuntu-${RELEASE}/amd64/initrd.gz -- quiet

label mainmenu
    menu label ^Back to main menu
    kernel vesamenu.c32
